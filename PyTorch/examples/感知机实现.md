class Perceptron(torch.nn.Module):
    def __init__(self, input_dim=2):
        self.linear = torch.nn.Linear(input_dim, 1, bias=True)
        torch.nn.init.normal_(self.linear.weight, mean=0.0, std=0.1)
        torch.nn.init.constant_(self.linear.bias, 0.0)

    def forward(self, x):
        z = self.linear(x)
        y_hat = torch.sign(z).squeeze(-1)
        return y_hat
